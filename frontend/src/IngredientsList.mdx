# IngredientsList

import IngredientsList from "./IngredientsList"
import { Playground, Props } from 'docz'
import Highlighter from "react-highlight-words";

> Ce composant permet d'afficher la liste des ingrédients nécessaires avec leur quantité correspondante pour une recette donnée.

## Props
<Props of={IngredientsList} />


## Exemples d'utilisation

<Playground>
  {() => {
    const ingredients = [
      { ingredient: "pommes de terre", quantite: "1", unite: "kg" },
      { ingredient: "oeufs", quantite: "3", unite: "pièce(s)" },
      { ingredient: "vinaigre non balsamique", quantite: "1", unite: "cas" },
      { ingredient: "radis", quantite: "2", unite: "botte(s)" },
      { ingredient: "oignons bottes", quantite: "2", unite: "pièce(s)" },
      { ingredient: "yaourt grec", quantite: "1", unite: "pièce(s)" },
      { ingredient: "mayonnaise", quantite: "1", unite: "cas" },
      { ingredient: "moutarde", quantite: "0.5", unite: "cas" },
      { ingredient: "ail", quantite: "1", unite: "gousse(s)" },
    ];
    const highlight= (texte) => texte;
    return (
      <IngredientsList
        ingredients={ingredients}
        highlight={highlight}
      />
    )
  }}
</Playground>

### Exemple avec le highlight

<Playground>
  {() => {
    const ingredients = [
      { ingredient: "pommes de terre", quantite: "1", unite: "kg" },
      { ingredient: "oeufs", quantite: "3", unite: "pièce(s)" },
      { ingredient: "vinaigre non balsamique", quantite: "1", unite: "cas" },
      { ingredient: "radis", quantite: "2", unite: "botte(s)" },
      { ingredient: "oignons bottes", quantite: "2", unite: "pièce(s)" },
      { ingredient: "yaourt grec", quantite: "1", unite: "pièce(s)" },
      { ingredient: "mayonnaise", quantite: "1", unite: "cas" },
      { ingredient: "moutarde", quantite: "0.5", unite: "cas" },
      { ingredient: "ail", quantite: "1", unite: "gousse(s)" },
    ];
    const highlight = (texte) => {
      return (
        <Highlighter
          highlightClassName="searchHighlight"
          searchWords={["yaourt", "mo"]}
          textToHighlight={texte}
        />
      );
    };
    return (
      <IngredientsList
        ingredients={ingredients}
        highlight={highlight}
      />
    )
  }}
</Playground>
