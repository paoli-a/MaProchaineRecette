# IngredientsCatalogue

import IngredientsCatalogue from "./IngredientsCatalogue";
import { Playground, Props } from "docz";

> Ce composant permet d'afficher les ingrédients du catalogue, d'en ajouter
> et d'en supprimer. Une recherche peut etre faite sur le nom des ingrédients et
> permettra d'afficher au fur et à mesure les ingrédients correspondants.

## Props

<Props of={IngredientsCatalogue} />

## Exemples d'utilisation

<Playground>
  {() => {
    const [ingredientsCatalogue, setIngredientsCatalogue] = React.useState([
      {
        nom: "Fraises",
      },
      {
        nom: "Sucre",
      },
      {
        nom: "Framboises",
      },
    ]);
    return (
      <IngredientsCatalogue
        possibleIngredients={ingredientsCatalogue}
        updatePossibleIngredients={setIngredientsCatalogue}
      />
    );
  }}
</Playground>
