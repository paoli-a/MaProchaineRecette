# FridgeIngredients

import FridgeIngredients from "./FridgeIngredients";
import { Playground, Props } from "docz";

> Ce composant permet d'afficher les ingrédients du frigo, d'en ajouter
> et d'en supprimer. Seul des noms d'ingrédients déjà présents dans le catalogue
> des ingrédients sont acceptés.

## Props

<Props of={FridgeIngredients} />

## Exemples d'utilisation

<Playground>
  {() => {
    const catalogIngredients = [
      {
        name: "Fraises",
      },
      {
        name: "Sucre",
      },
      {
        name: "Framboises",
      },
    ];
    const fridgeIngredients = [
      {
        id: 1,
        name: "Epinard",
        datePeremption: new Date(2020, 4, 15),
        quantite: "60",
        unit: "g",
      },
      {
        id: 2,
        name: "Céleri rave",
        datePeremption: new Date(2020, 3, 13),
        quantite: "1",
        unit: "g",
      },
    ];
    const units = ["kg", "g", "cl", "pièce(s)"];
    return (
      <FridgeIngredients
        ingredients={fridgeIngredients}
        possibleIngredients={catalogIngredients}
        totalUnits={units}
      />
    );
  }}
</Playground>
