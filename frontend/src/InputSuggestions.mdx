# InputSuggestions

import InputSuggestions from "./InputSuggestions";
import { Playground, Props } from "docz";
import { useForm } from "react-hook-form";

> Ce composant est un input permettant d'afficher une liste de suggestions personnalisée.

## Props

<Props of={InputSuggestions} />

Toutes les props supplémentaires seront transférées à la balise interne input.

## Exemples d'utilisation

<Playground>
  {() => {
    const possibleIngredients = [
      {
        name: "Fraises",
      },
      {
        name: "Sucre",
      },
      {
        name: "Framboises",
      },
    ];
    const [ingredientName, setIngredientName] = React.useState("");
    return (
      <form>
        <label htmlFor="ingredientName">Nom de l'ingrédient : </label>
        <InputSuggestions
          elements={possibleIngredients}
          id="ingredientName"
          getElementText={(ingredient) => ingredient.name}
          onChangeValue={(value) => setIngredientName(value)}
          value={ingredientName}
          name="ingredient"
          type="text"
        />
      </form>
    );
  }}
</Playground>
